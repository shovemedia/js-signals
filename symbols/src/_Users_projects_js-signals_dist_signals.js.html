 30     /**
 31      * Object that represents a binding between a Signal and a listener function.
 32      * <br />- <strong>This is an internal constructor and shouldn't be called by regular users.</strong>
 33      * <br />- inspired by Joa Ebert AS3 SignalBinding and Robert Penner's Slot classes.
 34      * @author Miller Medeiros
 35      * @constructor
 36      * @internal
 37      * @name signals.SignalBinding
 38      * @param {signals.Signal} signal Reference to Signal object that listener is currently bound to.
 39      * @param {Function} listener Handler function bound to the signal.
 40      * @param {boolean} isOnce If binding should be executed just once.
 41      * @param {Object} [listenerContext] Context on which listener will be executed (object that should represent the `this` variable inside listener function).
 42      * @param {Number} [priority] The priority level of the event listener. (default = 0).
 43      */
 44     function SignalBinding(signal, listener, isOnce, listenerContext, priority) {
 45 
 46         /**
 47          * Handler function bound to the signal.
 48          * @type Function
 49          * @private
 50          */
 51         this._listener = listener;
 52 
 53         /**
 54          * If binding should be executed just once.
 55          * @type boolean
 56          * @private
 57          */
 58         this._isOnce = isOnce;
 59 
 60         /**
 61          * Context on which listener will be executed (object that should represent the `this` variable inside listener function).
 62          * @memberOf signals.SignalBinding.prototype
 63          * @name context
 64          * @type Object|undefined|null
 65          */
 66         this.context = listenerContext;
 67 
 68         /**
 69          * Reference to Signal object that listener is currently bound to.
 70          * @type signals.Signal
 71          * @private
 72          */
 73         this._signal = signal;
 74 
 75         /**
 76          * Listener priority
 77          * @type Number
 78          * @private
 79          */
 80         this._priority = priority || 0;
 81     }
 82 
 83     SignalBinding.prototype = /** @lends signals.SignalBinding.prototype */ {
 84 
 85         /**
 86          * If binding is active and should be executed.
 87          * @type boolean
 88          */
 89         active : true,
 90 
 91         /**
 92          * Default parameters passed to listener during `Signal.dispatch` and `SignalBinding.execute`. (curried parameters)
 93          * @type Array|null
 94          */
 95         params : null,
 96 
 97         /**
 98          * Call listener passing arbitrary parameters.
 99          * <p>If binding was added using `Signal.addOnce()` it will be automatically removed from signal dispatch queue, this method is used internally for the signal dispatch.</p>
100          * @param {Array} [paramsArr] Array of parameters that should be passed to the listener
101          * @return {*} Value returned by the listener.
102          */
103         execute : function (paramsArr) {
104             var handlerReturn, params;
105             if (this.active && !!this._listener) {
106                 params = this.params? this.params.concat(paramsArr) : paramsArr;
107                 handlerReturn = this._listener.apply(this.context, params);
108                 if (this._isOnce) {
109                     this.detach();
110                 }
111             }
112             return handlerReturn;
113         },
114 
115         /**
116          * Detach binding from signal.
117          * - alias to: mySignal.remove(myBinding.getListener());
118          * @return {Function|null} Handler function bound to the signal or `null` if binding was previously detached.
119          */
120         detach : function () {
121             return this.isBound()? this._signal.remove(this._listener, this.context) : null;
122         },
123 
124         /**
125          * @return {Boolean} `true` if binding is still bound to the signal and have a listener.
126          */
127         isBound : function () {
128             return (!!this._signal && !!this._listener);
129         },
130 
131         /**
132          * @return {Function} Handler function bound to the signal.
133          */
134         getListener : function () {
135             return this._listener;
136         },
137 
138         /**
139          * Delete instance properties
140          * @private
141          */
142         _destroy : function () {
143             delete this._signal;
144             delete this._listener;
145             delete this.context;
146         },
147 
148         /**
149          * @return {boolean} If SignalBinding will only be executed once.
150          */
151         isOnce : function () {
152             return this._isOnce;
153         },
154 
155         /**
156          * @return {string} String representation of the object.
157          */
158         toString : function () {
159             return '[SignalBinding isOnce:' + this._isOnce +', isBound:'+ this.isBound() +', active:' + this.active + ']';
160         }
161 
162     };

170 	function validateListener(listener, fnName) {
171 		if (typeof listener !== 'function') {
172 			throw new Error( 'listener is a required param of {fn}() and should be a Function.'.replace('{fn}', fnName) );
173 		}

176 	/**
177 	 * Custom event broadcaster
178 	 * <br />- inspired by Robert Penner's AS3 Signals.
179 	 *
180 	 * @author Miller Medeiros
181 	 * @author <br />Jon Williams (type-checking enhancement)
182 	 * 
183 	* @param {...*} [params] can be:
184 	* <br />null (no type-checking), 
185 	* <br />OR a list of arguments matching the types of arguments to be dispatched, 
186 	* <br />OR a named hash of arguments if dispatching a named hash single object argument.
187 	 * 
188 	 * @constructor
189 	 * 
190 	 *
191 	 */
192 	signals.Signal = function () {
193 		/**
194 		 * @type Array.<SignalBinding>
195 		 * @private
196 		 */
197 		 
198 		var args = Array.prototype.slice.call(arguments);
199 				
200 		if (args.length > 1 || (args.length > 0 && !(isPlainObject(args[0]))) )
201 		{
202 			   //use standard arguments 'array'
203 			   this._argumentTypes = args;
204 			   this._argumentIsHash = false;
205 		}
206 		else if (args.length == 1 && (args[0] instanceof Object) )
207 		{
208 			   //use single-argument named hash
209 			   this._argumentTypes = arguments[0];
210 			   this._argumentIsHash = true;
211 		}
212 			   
213 				 
214 		this._bindings = [];
215 		this._prevParams = null;
216 	};

218 	signals.Signal.prototype = {
219 
220 		/**
221 		 * If Signal should keep record of previously dispatched parameters and
222 		 * automatically execute listener during `add()`/`addOnce()` if Signal was
223 		 * already dispatched before.
224 		 * @type boolean
225 		 */
226 		memorize : false,
227 
228 		/**
229 		 * @type boolean
230 		 * @private
231 		 */
232 		_shouldPropagate : true,
233 
234 		/**
235 		 * If Signal is active and should broadcast events.
236 		 * <p><strong>IMPORTANT:</strong> Setting this property during a dispatch will only affect the next dispatch, if you want to stop the propagation of a signal use `halt()` instead.</p>
237 		 * @type boolean
238 		 */
239 		active : true,
240 
241 		/**
242 		 * @param {Function} listener
243 		 * @param {boolean} isOnce
244 		 * @param {Object} [listenerContext]
245 		 * @param {Number} [priority]
246 		 * @return {SignalBinding}
247 		 * @private
248 		 */
249 		_registerListener : function (listener, isOnce, listenerContext, priority) {
250 
251 			var prevIndex = this._indexOfListener(listener, listenerContext),
252 				binding;
253 
254 			if (prevIndex !== -1) {
255 				binding = this._bindings[prevIndex];
256 				if (binding.isOnce() !== isOnce) {
257 					throw new Error('You cannot add'+ (isOnce? '' : 'Once') +'() then add'+ (!isOnce? '' : 'Once') +'() the same listener without removing the relationship first.');
258 				}
259 			} else {
260 				binding = new SignalBinding(this, listener, isOnce, listenerContext, priority);
261 				this._addBinding(binding);
262 			}
263 
264 			if(this.memorize && this._prevParams){
265 				binding.execute(this._prevParams);
266 			}
267 
268 			return binding;
269 		},
270 
271 		/**
272 		 * @param {SignalBinding} binding
273 		 * @private
274 		 */
275 		_addBinding : function (binding) {
276 			//simplified insertion sort
277 			var n = this._bindings.length;
278 			do { --n; } while (this._bindings[n] && binding._priority <= this._bindings[n]._priority);
279 			this._bindings.splice(n + 1, 0, binding);
280 		},
281 
282 		/**
283 		 * @param {Function} listener
284 		 * @return {number}
285 		 * @private
286 		 */
287 		_indexOfListener : function (listener, context) {
288 			var n = this._bindings.length,
289 				cur;
290 			while (n--) {
291 				cur = this._bindings[n];
292 				if (cur._listener === listener && cur.context === context) {
293 					return n;
294 				}
295 			}
296 			return -1;
297 		},
298 
299 		/**
300 		 * Check if listener was attached to Signal.
301 		 * @param {Function} listener
302 		 * @param {Object} [context]
303 		 * @return {boolean} if Signal has the specified listener.
304 		 */
305 		has : function (listener, context) {
306 			return this._indexOfListener(listener, context) !== -1;
307 		},

309 		/**
310 		 * Add a listener to the signal.
311 		 * @param {Function} listener Signal handler function.
312 		 * @param {Object} [listenerContext] Context on which listener will be executed (object that should represent the `this` variable inside listener function).
313 		 * @param {Number} [priority] The priority level of the event listener. Listeners with higher priority will be executed before listeners with lower priority. Listeners with same priority level will be executed at the same order as they were added. (default = 0)
314 		 * @return {SignalBinding} An Object representing the binding between the Signal and listener.
315 		 */
316 		add : function (listener, listenerContext, priority) {
317 			validateListener(listener, 'add');
318 			return this._registerListener(listener, false, listenerContext, priority);
319 		},
320 
321 		/**
322 		 * Add listener to the signal that should be removed after first execution (will be executed only once).
323 		 * @param {Function} listener Signal handler function.
324 		 * @param {Object} [listenerContext] Context on which listener will be executed (object that should represent the `this` variable inside listener function).
325 		 * @param {Number} [priority] The priority level of the event listener. Listeners with higher priority will be executed before listeners with lower priority. Listeners with same priority level will be executed at the same order as they were added. (default = 0)
326 		 * @return {SignalBinding} An Object representing the binding between the Signal and listener.
327 		 */
328 		addOnce : function (listener, listenerContext, priority) {
329 			validateListener(listener, 'addOnce');
330 			return this._registerListener(listener, true, listenerContext, priority);
331 		},
332 
333 		/**
334 		 * Remove a single listener from the dispatch queue.
335 		 * @param {Function} listener Handler function that should be removed.
336 		 * @param {Object} [context] Execution context (since you can add the same handler multiple times if executing in a different context).
337 		 * @return {Function} Listener handler function.
338 		 */
339 		remove : function (listener, context) {
340 			validateListener(listener, 'remove');
341 
342 			var i = this._indexOfListener(listener, context);
343 			if (i !== -1) {
344 				this._bindings[i]._destroy(); //no reason to a SignalBinding exist if it isn't attached to a signal
345 				this._bindings.splice(i, 1);
346 			}
347 			return listener;
348 		},
349 
350 		/**
351 		 * Remove all listeners from the Signal.
352 		 */
353 		removeAll : function () {
354 			var n = this._bindings.length;
355 			while (n--) {
356 				this._bindings[n]._destroy();
357 			}
358 			this._bindings.length = 0;
359 		},
360 
361 		/**
362 		 * @return {number} Number of listeners attached to the Signal.
363 		 */
364 		getNumListeners : function () {
365 			return this._bindings.length;
366 		},
367 
368 		/**
369 		 * Stop propagation of the event, blocking the dispatch to next listeners on the queue.
370 		 * <p><strong>IMPORTANT:</strong> should be called only during signal dispatch, calling it before/after dispatch won't affect signal broadcast.</p>
371 		 * @see signals.Signal.prototype.disable
372 		 */
373 		halt : function () {
374 			this._shouldPropagate = false;
375 		},
376 
377 		/**
378 		 * Dispatch/Broadcast Signal to all listeners added to the queue.
379 		 * @param {...*} [params] Parameters that should be passed to each handler.
380 		 * @throws Error if parameters do not match the signature provided to the Signal constructor.
381 		 */
382 		dispatch : function (params) {
383 			if (! this.active) {
384 				return;
385 			}
386 
387 			var paramsArr = Array.prototype.slice.call(arguments),
388 				n = this._bindings.length,
389 				bindings;
390 
391 				
392 			// strict parameter type checking
393 			
394 			if (this._argumentTypes)
395 			{	    
396 				// single argument named hash
397 				if (this._argumentIsHash)
398 				{
399 					if (paramsArr.length != 1)
400 					{
401 						throw ( new Error ('strict Signal argument length Mismatch. Expected single argument, named hash. Got: ' + paramsArr.length));
402 					}
403 					
404 					var signalData = paramsArr[0];
405 					
406 					for (var i in this._argumentTypes)
407 					{
408 						var value = signalData[i];
409 						var expectedType = this._argumentTypes[i];
410 						
411 						verifyType(value, expectedType, i);					 
412 					}   
413 				}
414 				else
415 				{
416 					// standard arguments 'array'
417 					if (paramsArr.length != this._argumentTypes.length)
418 					{
419 						throw ( new Error ('strict Signal argument length Mismatch. Expected: ' + this._argumentTypes.length + '. Got: ' + paramsArr.length));
420 					}
421 					
422 					for (var i=0, len=this._argumentTypes.length; i<len; i++)
423 					{
424 						var value = paramsArr[i];
425 						var expectedType = this._argumentTypes[i];
426 						
427 						verifyType(value, expectedType, i);					 
428 					}				   
429 				}
430 			}	
431 
432 			if (this.memorize) {
433 				this._prevParams = paramsArr;
434 			}
435 
436 			if (! n) {
437 				//should come after memorize
438 				return;
439 			}
440 
441 			bindings = this._bindings.slice(); //clone array in case add/remove items during dispatch
442 			this._shouldPropagate = true; //in case `halt` was called before dispatch or during the previous dispatch.
443 
444 			//execute all callbacks until end of the list or until a callback returns `false` or stops propagation
445 			//reverse loop since listeners with higher priority will be added at the end of the list
446 			do { n--; } while (bindings[n] && this._shouldPropagate && bindings[n].execute(paramsArr) !== false);
447 		},
448 
449 		/**
450 		 * Forget memorized arguments.
451 		 * @see signals.Signal.memorize
452 		 */
453 		forget : function(){
454 			this._prevParams = null;
455 		},
456 
457 		/**
458 		 * Remove all bindings from signal and destroy any reference to external objects (destroy Signal object).
459 		 * <p><strong>IMPORTANT:</strong> calling any method on the signal instance after calling dispose will throw errors.</p>
460 		 */
461 		dispose : function () {
462 			this.removeAll();
463 			delete this._bindings;
464 			delete this._prevParams;
465 		},

468 		 * @return {string} String representation of the object.
469 		 */
470 		toString : function () {
471 			return '[Signal active:'+ this.active +' numListeners:'+ this.getNumListeners() +']';
472 		}

476 	if (typeof jQuery != 'undefined') {
477 		var hasOwn = jQuery.hasOwn,
478 		isArray = jQuery.isArray,
479 		isNumeric = jQuery.isNumeric,
480 		isFunction = jQuery.isFunction,
481 		isWindow = jQuery.isWindow,
482 		isPlainObject = jQuery.isPlainObject
483 	}
484 	else
485 	{
486 		var objectToString = Object.prototype.toString,
487 		
488 		hasOwn = Object.prototype.hasOwnProperty,
489 		
490 		isArray = (Array.isArray) || function(obj) {
491 			return objectToString.call(obj) == '[object Array]';
492 		},
493 		
494 		isNumeric = function(a) {
495 			return !isNaN(parseFloat(a)) && isFinite(a);
496 		},
497 		
498 		
499 		isFunction = function( obj ) {
500 			return objectToString.call(obj) == '[object function]';
501 		},
502 	
503 		// A crude way of determining if an object is a window
504 		isWindow = function( obj ) {
505 			return obj && typeof obj === "object" && "setInterval" in obj;
506 		},
507 	
508 		isPlainObject = function( obj ) {
509 			// Must be an Object.
510 			// Because of IE, we also have to check the presence of the constructor property.
511 			// Make sure that DOM nodes and window objects don't pass through, as well
512 			
513 			if ( !obj || objectToString.call(obj) !== "[object Object]" || obj.nodeType || isWindow( obj ) ) {
514 				return false;
515 			}
516 	
517 			try {
518 				// Not own constructor property must be Object
519 				if ( obj.constructor &&
520 					!hasOwn.call(obj, "constructor") &&
521 					!hasOwn.call(obj.constructor.prototype, "isPrototypeOf") ) {
522 					return false;
523 				}
524 			} catch ( e ) {
525 				// IE8,9 Will throw exceptions on certain host objects #9897
526 				return false;
527 			}
528 	
529 			// Own properties are enumerated firstly, so to speed up,
530 			// if last one is own, then all properties are own.
531 	
532 			var key;
533 			for ( key in obj ) {}
534 	
535 			if ( key === undefined || hasOwn.call( obj, key ))
536 			{
537 				return true
538 			}
539 			else
540 			{
541 				return false
542 			};
543 		}		
544 	}

548 	var verifyType = function (value, expectedType, i)
549 	{
550 		switch (expectedType)
551 		{
552 			case Number:
553 				if (!(isNumeric(value) && !isArray(value) && !(value.constructor === String || value instanceof String)))
554 				{
555 					throw ( new Error ('strict Signal argument Type Mismatch on argument ' + i + ' is not Number'));
556 				}
557 			break;
558 			case String:
559 				if (!(value.constructor === String || value instanceof String))
560 				{
561 					throw ( new Error ('strict Signal argument Type Mismatch on argument ' + i + ' is not String'));
562 				}
563 			break;
564 			case Array:
565 				if (!isArray(value))
566 				{
567 					throw ( new Error ('strict Signal argument Type Mismatch on argument ' + i + ' is not Array'));							 
568 				}
569 			break;
570 			default:
571 				if (!(value instanceof expectedType || expectedType.isPrototypeOf(value)) )
572 				{
573 					throw ( new Error ('strict Signal argument Type Mismatch on argument ' + i + ' is not custom Class type: ' + expectedType));
574 				}
575 				break;
576 		}	   
577 	}
